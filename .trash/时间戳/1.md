// 国内通用规则：周一为一周起始，当月1号所在周为第1周
const nowMoment = moment();
// 获取当月1号的日期对象
const monthStart = nowMoment.clone().startOf('month');
// 计算本月周数，处理跨年/跨月边界异常
const weekDiff = nowMoment.isoWeek() - monthStart.isoWeek();
const weekOfMonth = weekDiff < 0 ? weekDiff + 52 : weekDiff;
const finalWeekNum = weekOfMonth + 1;
// 最终显示的文本
const 本月周数 = `本月第${finalWeekNum}周`;
